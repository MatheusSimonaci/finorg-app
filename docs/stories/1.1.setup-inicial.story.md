## Dev Notes

### Schema Prisma inicial

```prisma
model Account {
  id          String        @id @default(cuid())
  name        String
  institution String        // nubank | xp | bitybank | outro
  type        String        // checking | investment | crypto
  transactions Transaction[]
  assets      Asset[]
  createdAt   DateTime      @default(now())
}

model Transaction {
  id                     String    @id @default(cuid())
  date                   DateTime
  description            String
  amount                 Decimal
  accountId              String
  account                Account   @relation(fields: [accountId], references: [id])
  nature                 String?   // pessoal | empresa | work_tool | misto
  category               String?
  subcategory            String?
  type                   String?   // gasto | investimento | reserva | receita | transferencia
  confidence             Float?
  reasoning              String?
  isReimbursable         Boolean   @default(false)
  classificationOverride Boolean   @default(false)
  importBatchId          String?
  hash                   String    @unique
  createdAt              DateTime  @default(now())
}

model ClassificationRule {
  id          String   @id @default(cuid())
  pattern     String
  nature      String
  category    String
  type        String?
  subcategory String?
  source      String   @default("user") // user | system
  createdAt   DateTime @default(now())
}

model Asset {
  id           String   @id @default(cuid())
  name         String
  type         String   // tesouro | cdb | lci | fii | acoes | cripto | previdencia | fundo | conta_remunerada
  subtype      String?
  institution  String
  currentValue Decimal  @default(0)
  lastUpdated  DateTime @default(now())
  purpose      String   @default("personal") // personal | business | reserve | dream
  dreamId      String?
  dream        Dream?   @relation(fields: [dreamId], references: [id])
  active       Boolean  @default(true)
  createdAt    DateTime @default(now())
}

model Dream {
  id             String    @id @default(cuid())
  name           String
  targetAmount   Decimal
  targetDate     DateTime?
  priorityOrder  Int
  status         String    @default("planejando") // planejando | acumulando | realizado | arquivado
  achievedAt     DateTime?
  archivedAt     DateTime?
  notes          String?
  assets         Asset[]
  createdAt      DateTime  @default(now())
}

model Budget {
  id                 String   @id @default(cuid())
  category           String
  targetPct          Float
  alertThresholdPct  Float    @default(80)
  monthOverride      String?  // YYYY-MM, null = todos os meses
  createdAt          DateTime @default(now())
  updatedAt          DateTime @updatedAt
}

model EmergencyReserveConfig {
  id                       String   @id @default(cuid())
  targetMonths             Int      @default(4)
  calculationWindowMonths  Int      @default(3)
  excludeOutliers          Boolean  @default(false)
  createdAt                DateTime @default(now())
  updatedAt                DateTime @updatedAt
}
```
