## Dev Notes

### Tipos de alerta

```typescript
type Alert = {
  id: string
  type: 'budget_warning' | 'budget_exceeded' | 'upcoming_installment' | 'seasonal_expense'
  severity: 'warning' | 'critical'
  title: string
  message: string
  category?: string
  amount?: number
  dueDate?: Date
  snoozedUntil?: Date
}
```

### Cálculo de alertas (executado em cada carregamento do dashboard)

```typescript
function computeAlerts(budgetSummary, recurringExpenses, personalIncome): Alert[] {
  const alerts: Alert[] = []

  // Budget warnings
  for (const [category, data] of Object.entries(budgetSummary)) {
    if (data.pct >= 80 && data.pct < 100) alerts.push({ type: 'budget_warning', severity: 'warning', category })
    if (data.pct >= 100) alerts.push({ type: 'budget_exceeded', severity: 'critical', category })
  }

  // Parcelas e sazonais no próximo mês
  const upcoming = recurringExpenses.filter(e => isDueNextMonth(e))
  for (const expense of upcoming) {
    alerts.push({ type: 'upcoming_installment', severity: 'warning', ...expense })
  }

  return alerts.filter(a => !isSnoozed(a))
}
```
