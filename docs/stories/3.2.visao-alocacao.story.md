## Dev Notes

### Estados da carteira

```typescript
type PortfolioState = 'EQUILIBRADA' | 'SOB_SONHO' | 'REBALANCEANDO'

function computePortfolioState(allocation: AllocationTarget[], activeDream: Dream | null): PortfolioState {
  if (!activeDream) return checkBalance(allocation) ? 'EQUILIBRADA' : 'REBALANCEANDO'
  if (activeDream.status === 'acumulando') return 'SOB_SONHO'
  return 'REBALANCEANDO'
}
```
