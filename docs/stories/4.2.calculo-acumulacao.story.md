## Dev Notes

### Cálculo de meses restantes

```typescript
function monthsUntil(targetDate: Date): number {
  const now = new Date()
  return (targetDate.getFullYear() - now.getFullYear()) * 12 
       + (targetDate.getMonth() - now.getMonth())
}

function calculateRequiredMonthlyContribution(dream: Dream, earmarked: number): number {
  const remaining = Number(dream.targetAmount) - earmarked
  const months = monthsUntil(dream.targetDate!)
  if (months <= 0) return remaining // vencido
  return remaining / months
}
```
